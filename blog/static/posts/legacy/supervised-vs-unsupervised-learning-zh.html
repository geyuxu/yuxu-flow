import torch
from torch import nn

# å®šä¹‰ç”Ÿæˆå™¨
class Generator(nn.Module):
    def __init__(self, z_dim=100, img_dim=784):
        super().__init__()
        self.net = nn.Sequential(
            nn.Linear(z_dim, 256),
            nn.ReLU(True),
            nn.Linear(256, 512),
            nn.ReLU(True),
            nn.Linear(512, img_dim),
            nn.Tanh()  # å°†è¾“å‡ºå½’ä¸€åŒ–åˆ°[-1, 1]
        )
    def forward(self, z):
        return self.net(z)

# å®šä¹‰åˆ¤åˆ«å™¨
class Discriminator(nn.Module):
    def __init__(self, img_dim=784):
        super().__init__()
        self.net = nn.Sequential(
            nn.Linear(img_dim, 512),
            nn.LeakyReLU(0.2, inplace=True),
            nn.Linear(512, 256),
            nn.LeakyReLU(0.2, inplace=True),
            nn.Linear(256, 1),
            nn.Sigmoid() # è¾“å‡ºä¸€ä¸ª[0, 1]çš„æ¦‚ç‡å€¼
        )
    def forward(self, x):
        return self.net(x)

# åˆå§‹åŒ–æ¨¡å‹ã€ä¼˜åŒ–å™¨å’ŒæŸå¤±å‡½æ•°
G = Generator()
D = Discriminator()
g_opt = torch.optim.Adam(G.parameters(), lr=2e-4)
d_opt = torch.optim.Adam(D.parameters(), lr=2e-4)
criterion = nn.BCELoss()

print("GAN components initialized successfully.")
